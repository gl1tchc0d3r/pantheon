### Step 4: Agent Loop Module

Create `src/agent.rs`:

```rust
pub struct AgentLoop {
    provider: Box<dyn LlmProvider>,
}

impl AgentLoop {
    pub fn new(provider: Box<dyn LlmProvider>) -> Self {
        Self { provider }
    }

    pub async fn run(&self, input: &str) -> Result<String, AgentError> {
        // Build simple prompt
        let prompt = format!("User: {}\nAo:", input);
        
        // Call LLM
        let response = self.provider.complete(&prompt).await?;
        
        Ok(response)
    }
}
```

**Deliverable**: Agent that takes input, calls LLM, returns response.

### Step 5: TUI Module

Create `src/tui/mod.rs`:

```rust
pub struct Tui {
    input_mode: bool,
    messages: Vec<Message>,
}

impl Tui {
    pub fn new() -> Self {
        Self {
            input_mode: true,
            messages: Vec::new(),
        }
    }

    pub fn add_message(&mut self, role: &str, content: &str) {
        self.messages.push(Message { role: role.to_string(), content: content.to_string() });
    }

    pub fn handle_input(&self, input: &str) -> InputResult {
        match input.trim() {
            "/quit" => InputResult::Quit,
            "/help" => InputResult::Command("help".to_string()),
            _ => InputResult::Chat(input.to_string()),
        }
    }
}
```

**Deliverable**: TUI struct with message handling.

### Step 6: Main Entry Point

Create `src/main.rs`:

```rust
mod config;
mod provider;
mod agent;
mod tui;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Initialize logging
    tracing_subscriber::fmt::init();
    
    // Load config
    let config = config::Config::load()?;
    
    // Initialize provider
    let provider = OpenRouterProvider::new(
        config.provider.api_key,
        config.provider.model,
    );
    
    // Initialize agent loop
    let agent = AgentLoop::new(Box::new(provider));
    
    // Initialize TUI
    let mut tui = Tui::new();
    
    // Main loop
    loop {
        let input = read_line();
        
        match tui.handle_input(&input) {
            InputResult::Quit => break,
            InputResult::Command("help") => {
                print_help();
            }
            InputResult::Chat(text) => {
                tui.add_message("user", &text);
                
                let response = agent.run(&text).await?;
                tui.add_message("ao", &response);
                
                print_messages(&tui.messages);
            }
        }
    }
    
    Ok(())
}
```

**Deliverable**: Working application that accepts input, calls LLM, displays response.
